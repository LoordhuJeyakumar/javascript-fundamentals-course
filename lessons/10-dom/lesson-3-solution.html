// Example: remove delegated listener
const buttons = document.getElementById('buttons');
function handler(e){ const btn = e.target.closest('button'); if(!btn) return; console.log('clicked', btn.dataset.index); }
buttons.addEventListener('click', handler);
// later...
buttons.removeEventListener('click', handler);

// Capturing example
document.getElementById('buttons').addEventListener('click', (e)=>console.log('capturing'), true);
