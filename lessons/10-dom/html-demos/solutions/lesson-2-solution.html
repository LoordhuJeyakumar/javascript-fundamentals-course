Example implementations for the exercises.

// removeItem implementation
function removeItem(index){
  const list = document.getElementById('list');
  const item = list.children[index];
  if(item) list.removeChild(item);
}

// insertBefore implementation
function insertBeforeSecond(text){
  const list = document.getElementById('list');
  const li = document.createElement('li');
  li.textContent = text;
  if(list.children[1]){
    list.insertBefore(li, list.children[1]);
  } else {
    list.appendChild(li);
  }
}

// replaceChild implementation
function replaceSecond(text){
  const list = document.getElementById('list');
  const li = document.createElement('li');
  li.textContent = text;
  if(list.children[1]){
    list.replaceChild(li, list.children[1]);
  }
}
